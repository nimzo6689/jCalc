24  package com.qiita.nimzo6689.calculator.code;
25  
26  import javafx.scene.control.TextArea;
27  import lombok.AccessLevel;
28  import lombok.Getter;
29  import lombok.RequiredArgsConstructor;
30  
31  /**
32   *
33   * @author nimzo6689
34   */
35  @Getter
36  @RequiredArgsConstructor(access = AccessLevel.PRIVATE)
37  public enum CalcNumber {
38  
39      COMMA("comma", "."),
40      ZERO("zero", "0"),
41      DOUBLE_ZERO("double_zero", "00"),
42      ONE("one", "1"),
43      TWO("two", "2"),
44      THREE("three", "3"),
45      FOUR("four", "4"),
46      FIVE("five", "5"),
47      SIX("six", "6"),
48      SEVEN("seven", "7"),
49      EIGHT("eight", "8"),
50      NINE("nine", "9"),;
51  
52      /**
53       *
54       * @param value
55       * @return
56       */
57      public static CalcNumber of(String value) {
58          for (CalcNumber calcNumber : CalcNumber.values()) {
59              if (calcNumber.value.equals(value)) {
60                  return calcNumber;
61              }
62          }
63          throw new IllegalArgumentException(
64                  "Type:" + value + " is not a valid " + CalcNumber.class.getName() + " value.");
65      }
66  
67      private final String value;
68      private final String number;
69  
70      /**
71       *
72       * @param textArea
73       */
74      public void appendNumberTo(TextArea textArea) {
75          textArea.appendText(number);
76      }
77  
78  }
